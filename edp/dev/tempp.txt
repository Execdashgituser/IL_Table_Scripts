UPDATE ISSUES SET CRITICALITY_NUM = Case when UPPER(CRITICALITY) = 'GREEN' then 3 when UPPER(CRITICALITY) = 'YELLOW' then 2 when UPPER(CRITICALITY) = 'RED' then 1 else 0 End where BATCH_ID = (select MAX(BATCH_ID) from [dbo].[ISSUES]);
UPDATE ISSUES SET IS_DUE_NXT_6MON = Case when (DATEDIFF ( m , CURRENT_TIMESTAMP ,RESOLUTION_DUE_DATE) between 0 and 6) and   (DATEDIFF ( d , CURRENT_TIMESTAMP ,RESOLUTION_DUE_DATE)>= 0) then 1 else 0 End where BATCH_ID = (select MAX(BATCH_ID) from [dbo].[ISSUES]) ;
